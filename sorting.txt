struct node{
	int data;
	struct node* next;
};
struct node* add_next(struct node* tail, int data){
	struct node* ptr=malloc(sizeof(struct node));
	ptr->data=data;
	if(tail==NULL){
		tail=ptr;
		tail->next=ptr;
	}
	else{
		struct node* temp=tail->next;
		tail->next=ptr;
		tail=ptr;
		tail->next=temp;
	}
	return tail;
}
struct node* sort(struct node*tail){

	int temp2;

				temp2=temp->data;
				temp->data=temp1->data;
				temp1->data=temp2;
			}
			temp1=temp1->next;
		}while(temp1!=tail->next);
		temp=temp->next;
	}while(temp!=tail->next);
	return tail;
}
